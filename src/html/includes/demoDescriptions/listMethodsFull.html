1) this.status.whenReady: () => Promise<void>

Returns a promise and allows using then-callback when the list is in ready status. 


new InfinityScroll(props).status.whenReady()
  .then(() => {
    console.log(`List is ready!`);
    return {instance, success: true, error: null}
  })

2) this.status.waitForStatus: (targetStatus: Status) => Promise<Status>

Returns a promise and allows using then-callback when the list reaches the target status 


type Status = "initialization" | "loading data" | "ready" | "error"
  

new InfinityScroll(props).status.waitForStatus("loading data")
  .then(() => {
    console.log(`Now list is in loading data!`);
    return {instance, success: true, error: null}
  })
